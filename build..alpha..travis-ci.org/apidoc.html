function yuicompressor.compressString (str, options, callback)

  • description and source-code
    compressString = function (str, options, callback) {
    //Now we have a string, spawn and pipe it in.

    options = filterOptions(options);

    var args = [
        '-jar',
        exports.jar
    ], buffer = '', errBuffer = '', child;

    Object.keys(options).forEach(function(key) {
        args.push('--' + key);
        if (options[key] && options[key] !== true) {
            args.push(options[key]);
        }
    });

    child = spawn('java', args, {
        stdio: ['pipe', 'pipe', 'pipe']
    });

    child.stdin.write(str);
    child.stdin.end();

    child.stdout.on('data', function(chunk) {
        buffer += chunk;
    });
    child.stderr.on('data', function(chunk) {
        errBuffer += chunk;
    });

    child.on('exit', function() {
        var err = null;
        if (errBuffer.indexOf('[ERROR]') > -1) {
            err = errBuffer;
        }
        callback(err, buffer, errBuffer);
    });
}
